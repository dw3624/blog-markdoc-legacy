    # Rows that appear in both ydf and zdf (Intersection)

(pd.merge(ydf, zdf, how = 'outer', indicator = True)
    .query('_merge == "left_only"')
    .drop(columns=['_merge']))
    # Rows that appear in ydf but not zdf (Setdiff)
    # zdf에는 없지만 ydf에는 있는 행 (차집합, ydf - zdf)
